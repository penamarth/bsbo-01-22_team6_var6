```plantuml
@startuml
' Определение классов
class Product {
    +int id
    +string name
    +string category
    +int quantity
    +float unitPrice
    +StorageLocation storageLocation
    +bool getAvailability()
    +void updateQuantity(int quantity)
}

class StorageLocation {
    +int id
    +string description
    +Warehouse warehouse
    +bool isFull()
    +void addProduct(Product product)
    +void removeProduct(Product product)
}

class Warehouse {
    +int id
    +string address
    +int totalCapacity
    +List<StorageLocation> storageLocations
    +List<StorageLocation> getAvailableLocations()
    +void addStorageLocation(StorageLocation location)
    +void UpdateLocation()
}

class Operation {
    +int id
    +DateTime date
    +User user
    +List<Operation> fetchOperations(DateTime startDate, DateTime endDate)
    +void execute()
}

class ShipmentOperation{
    +Shipment shipment
    +bool isSent
}

class ReceiptOperation{
    +Product product
}

class User {
    +int id
    +string name
    +string role

    +void performOperation(Operation operation)
}

class Manager {
    +void generateReport(DateTime startDate, DateTime endDate)
}

class Operator {
    +void processProductReceipt(Product product, int quantity)
    +void processProductDispatch(Product product, int quantity)
}

class Shelf{
    +int shelfNumber
}

class Cell{
    +int cellNumber
}

class Shipment{
    +List<Product> products
    +void sendShipment()
}


' Связи
Product --> StorageLocation : "1 *"
StorageLocation --> Warehouse : "1 *"
Operation --> Product : "1 *"
Operation --> User : "1 *"
User <|-- Manager
User <|-- Operator

StorageLocation <|-- Shelf
StorageLocation <|-- Cell

Operation <|-- ShipmentOperation
Operation <|-- ReceiptOperation

' Агрегация и композиция
Warehouse o-- StorageLocation : "contains"
Operation *-- Product : "related"
Shipment *-- Product : "related"

Shipment *-- ShipmentOperation : "related"


@enduml

```
