```plantuml
@startuml
' Определение классов
class Product {
    +int id
    +string name
    +string category
    +int quantity
    +float unitPrice
    +StorageLocation storageLocation
    +bool getAvailability()
    +void updateQuantity(int quantity)
}

class StorageLocation {
    +int id
    +string description
    +int capacity
    +Warehouse warehouse
    +bool isFull()
    +void addProduct(Product product)
}

class Warehouse {
    +int id
    +string address
    +int totalCapacity
    +List<StorageLocation> storageLocations
    +List<StorageLocation> getAvailableLocations()
    +void addStorageLocation(StorageLocation location)
}

class Operation {
    +int id
    +DateTime date
    +string type
    +Product product
    +int quantity
    +User user
    +void execute()
}

class User {
    +int id
    +string name
    +string role
    +void performOperation(Operation operation)
}

class Manager {
    +void generateReport(DateTime startDate, DateTime endDate)
}

class Operator {
    +void processProductReceipt(Product product, int quantity)
    +void processProductDispatch(Product product, int quantity)
}

' Связи
Product --> StorageLocation : "1 *"
StorageLocation --> Warehouse : "1 *"
Operation --> Product : "1 *"
Operation --> User : "1 *"
User <|-- Manager
User <|-- Operator

' Агрегация и композиция
Warehouse o-- StorageLocation : "contains"
Operation *-- Product : "related"

@enduml

```
